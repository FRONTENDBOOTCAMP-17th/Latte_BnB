> 리뷰 항목: 3-1

### 1단계: 파일 열기

### 2단계: 오타 수정 (25행)

### 3단계: 데이터 접근 방식 수정 (29~31행)

`rsv`는 배열이에요. 배열에서 `.accommodation`을 직접 꺼내면 `undefined`가 됩니다.

**수정 전:**
```js
const rsv = rsvData.data.reservations;
const amdData = rsv.accommodation;    // 이 두 줄 삭제!
const schData = rsv.schedule;         // 이 두 줄 삭제!
```

**수정 후:**
```js
const rsv = rsvData.data.reservations;
// amdData, schData 줄은 삭제합니다
```

### 4단계: 반복문 안에서 개별 항목 접근 (40~41행)

**수정 전:**
```js

  const amd = amdData[i];    // undefined[i] — 에러!
  const sch = schData[i];    // undefined[i] — 에러!
```

**수정 후:**
```js
for (let i = 0; i < rsv.length; i++) {
  const amd = rsv[i].accommodation;   // 각 예약의 숙소 정보
  const sch = rsv[i].schedule;        // 각 예약의 일정 정보
```

### 5단계: localStorage.setItem 수정 (63~64행)

`localStorage.setItem()`은 항상 **(키, 값)** 두 개의 인자가 필요해요.

**수정 전:**
```js
localStorage.setItem(rsv.id);       // 인자 1개 — 동작 안 함
localStorage.setItem(rsv.status);   // 인자 1개 — 동작 안 함
```

**수정 후:**
```js
localStorage.setItem('id', rsv[i].id);
localStorage.setItem('status', rsv[i].status);
```

> 참고: `rsv`가 아니라 `rsv[i]`여야 해요! `rsv`는 배열이라 `.id`가 없습니다.

### 6단계: style 오타 수정 (69행)

### 완성된 전체 코드

수정이 잘 됐는지 확인하려면 아래 전체 코드와 비교해보세요.

```js
import { buildHeader } from '../src/components/header.js';
import { buildFooter } from '../src/components/footer.js';
import constants from '../src/constants.js';

const API_BASE = constants.API_BASE_URL;

document.body.prepend(buildHeader());
document.body.append(buildFooter());

const rsvList = document.getElementById('reservation-list');
const result = document.getElementById('result');
const token = localStorage.getItem('accessToken');

async function loadReservation() {
  try {
    const res = await fetch(`${API_BASE}/me/reservations`, {
      method: 'GET',
      headers: {
        'Content-Type': 'application/json',
        Authorization: 'Bearer ' + token,
      },
    });

    if (!res.ok) {
      throw new Error('HTTP 오류: ' + res.status);
    }

    const rsvData = await res.json();
    const rsv = rsvData.data.reservations;

    if (rsv.length === 0) {
      result.textContent = '예약 내역이 없습니다.';
      return;
    }

    rsvList.textContent = '';

    for (let i = 0; i < rsv.length; i++) {
      const amd = rsv[i].accommodation;
      const sch = rsv[i].schedule;

      const resDiv = document.createElement('div');
      const cardDiv = document.createElement('div');
      cardDiv.className = 'reservation-card';
      const cardImg = document.createElement('img');
      cardImg.className = 'img';
      const cardName = document.createElement('p');
      cardName.className = 'name';
      const cardPeriod = document.createElement('p');

      cardImg.src = amd.thumbnailUrl;
      cardName.textContent = amd.title;
      cardPeriod.textContent = sch.checkInDate + '~' + sch.checkOutDate;

      cardDiv.appendChild(cardImg);
      cardDiv.appendChild(cardName);
      cardDiv.appendChild(cardPeriod);
      resDiv.appendChild(cardDiv);
      rsvList.appendChild(resDiv);

      localStorage.setItem('id', rsv[i].id);
      localStorage.setItem('status', rsv[i].status);
    }
  } catch (e) {
    result.textContent = '에러 발생: ' + e.message;
    result.style.color = 'red';
  }
}

loadReservation();
```

### 2단계: auth.js 작성

`src/utils/auth.js` 파일을 만들고 아래 내용을 넣으세요.

```js
import constants from '../constants.js';

/**
 * localStorage에서 토큰을 가져옵니다.
 */
export function getToken() {
  return localStorage.getItem('accessToken');
}

/**
 * 토큰이 유효한지 확인하고, 유효하면 사용자 데이터를 반환합니다.
 * 유효하지 않으면 null을 반환합니다.
 */
export async function checkAuth() {
  const token = getToken();
  if (!token) return null;

  try {
    const res = await fetch(`${constants.API_BASE_URL}/me/profile`, {
      method: 'GET',
      headers: {
        Authorization: `Bearer ${token}`,
      },
    });

    if (!res.ok) return null;

    const { data } = await res.json();
    return data;
  } catch (error) {
    console.error('인증 확인 실패:', error.message);
    return null;
  }
}

/**
 * 로그인이 반드시 필요한 페이지에서 사용합니다.
 * 인증에 실패하면 로그인 페이지로 이동시킵니다.
 *
 * 사용법:
 *   const user = await requireAuth();
 *   if (!user) return;  // 이미 로그인 페이지로 이동함
 */
export async function requireAuth() {
  const user = await checkAuth();
  if (!user) {
    alert('로그인이 필요합니다.');
    location.href = '/login/';
    return null;
  }
  return user;
}
```
